#include <iostream>

int height, weight = 0;
float bmi = 0;

int main(){
    std::cout << "Enter your height in inches: ";
    std::cin >> height;
    std::cout << "Enter your weight in pounds: ";
    std::cin >> weight;
    bmi = (weight*703)/(height*height);
    std::cout << "Your BMI is: " << bmi << "\nYou are ";

    if (bmi < 18.5){
        std::cout << "underweight.";
    }
    else if ((bmi >= 18.5) && (bmi <= 24.9)) {
        std::cout << "normal.";
    }
    else if ((bmi >= 25) && (bmi <= 29.9)) {
        std::cout << "overweight.";
    }
    else{
        std::cout << "obese.";
    }
}
