#include <iostream>
#include <math.h>

using namespace std;

int whole(int a, int b)
{
    return a/b;
}

int remainder(int a, int b){
    return a%b;
}

void formatnumber(int a){
    if (a < 10)
    {
        cout << a;
    }
    else {
        formatnumber(whole(a, 10));
        cout << "  " << remainder(a, 10);
    }
}

int main(){
    int seed = 0;
    cout << "Enter a seed: ";
    cin >> seed;
    srand(seed);
    int number = rand();
    cout << "Unformatted number: " << number << endl;
    formatnumber(number);
}

#include <iostream>

using namespace std;

int toMinutes(int days, int hours, int minutes){
    return days*24*60+hours*60+minutes;
}

int main(){
    int days = 0;
    int hours = 0;
    int minutes = 0;
    cout << "Enter days: ";
    cin >> days;
    cout << "Enter hours: ";
    cin >> hours;
    cout << "Enter minutes: ";
    cin >> minutes;

    int days2 = 0;
    int hours2 = 0;
    int minutes2 = 0;
    cout << "Enter days for 2nd time: ";
    cin >> days2;
    cout << "Enter hours for 2nd time: ";
    cin >> hours2;
    cout << "Enter minutes for 2nd time: ";
    cin >> minutes2;

    int time = toMinutes(days, hours, minutes);
    int time2 = toMinutes(days2, hours2, minutes2);

    int difference = 0;

    if (time > time2)
    {
        difference = time - time2;
    }
    else if (time2 > time)
    {
        difference = time2 - time;
    }

    cout << "Minutes between times: " << difference;
}

#include <iostream>
#include <iomanip>

using namespace std;

float toC(float f)
{
    float c = (f-32)*5/9;
    return c;
}

float toF(int c)
{
    float f = c*(9/5)+32;
    return f;
}

void cTable()
{
    for (int i = 0; i < 4; i++)
    {
        cout << setw(4) << "C";
        cout << setw(8) << "F";
        cout << setw(10) << " ";
    }
    for (int c = 0; c <=100; c++)
    {
        if (c%4==0)
        {
            cout << endl;
        }
        cout << setw(4) << c;
        cout << setw(8) << toF(c);
        cout << setw(10) << " ";
    }
    cout << endl << endl;
}

void fTable()
{
    for (int i = 0; i < 4; i++)
    {
        cout << setw(4) << "F";
        cout << setw(8) << "C";
        cout << setw(10) << " ";
    }
    for (int f = 32; f <=212; f++)
    {
        if (f%4==0)
        {
            cout << endl;
        }
        cout << setw(4) << f;
        cout << setw(8) << toC(f);
        cout << setw(10) << " ";
    }
    cout << endl << endl;
}
